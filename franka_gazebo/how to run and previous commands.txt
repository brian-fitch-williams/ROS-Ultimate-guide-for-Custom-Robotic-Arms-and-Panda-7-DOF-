## Launch the world with cup and robot with controllers 
roslaunch franka_gazebo panda_world.launch 
## to run the trajectory_command for the robot to ocome near cup in gazebo
rosrun franka_gazebo trajectory_command.py 0.8 0.4 .17 0.04 0.04  ## a bit up then the cup
rosrun franka_gazebo trajectory_command.py 0.8 0.4 .07 0.04 0.04  ## around the handle of cup
rosrun franka_gazebo trajectory_command.py 0.8 0.4 .07 0.0 0.0  ## closing the claws
rosrun franka_gazebo trajectory_command.py 0.8 0.4 1.2 0.0 0.0  ## Bring the glass to a hieght
